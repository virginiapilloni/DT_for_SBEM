        <script>
            function minutesToTime(minutes) {
                const hours = Math.floor(minutes / 60);
                const mins = minutes % 60;
                const seconds = 0; // Aggiungi i secondi se necessario
                return `${String(hours).padStart(2, '0')}:${String(mins).padStart(2, '0')}:${String(seconds).padStart(2, '0')}`;
            }
        
            function fetchData() {
                let coffeeData = [];
                let washingMachineData = [];
                let coffeeIndex = 0;
                let washingMachineIndex = 0;
        
                function updateValues() {
                    if (coffeeIndex < coffeeData.length || washingMachineIndex < washingMachineData.length) {
                        const coffeeItem = coffeeIndex < coffeeData.length ? coffeeData[coffeeIndex] : null;
                        const washingMachineItem = washingMachineIndex < washingMachineData.length ? washingMachineData[washingMachineIndex] : null;
        
                        let latestTime = '';
        
                        if (coffeeItem) {
                            const coffeeConsumption = coffeeItem.consumption_coffee;
                            const coffeeTime = coffeeItem.orario; // Orario in formato HH:MM:SS
        
                            document.getElementById('coffee-consumption').textContent = `Coffee Consumption: ${coffeeConsumption.toFixed(2)} kWh`;
                            document.getElementById('coffee-machine-consumption').textContent = coffeeConsumption.toFixed(2);
        
                            const coffeeMachineElement = document.getElementById('coffee-machine');
                            coffeeMachineElement.classList.toggle('on', coffeeConsumption > 0);
                            coffeeMachineElement.classList.toggle('off', coffeeConsumption <= 0);
        
                            // Aggiorna il tempo più recente
                            latestTime = coffeeTime;
                        }
        
                        if (washingMachineItem) {
                            const washingMachineConsumption = washingMachineItem.consumption_wm;
                            const washingMachineTime = minutesToTime(washingMachineItem.time); // Conversione dei minuti in HH:MM:SS
                            
                            // Mostra il valore di consumo della lavatrice con la descrizione
                            document.getElementById('washing-machine-consumption').textContent = ` ${washingMachineConsumption.toFixed(2)} `;
                            
                            // Aggiorna l'elemento dell'icona della lavatrice
                            const washingMachineElement = document.getElementById('washing-machine');
                            washingMachineElement.classList.toggle('on', washingMachineConsumption > 0);
                            washingMachineElement.classList.toggle('off', washingMachineConsumption <= 0);

                            // Aggiorna il tempo più recente solo se l'orario è più recente di quello attuale
                            if (latestTime === '' || washingMachineItem.orario > coffeeItem?.orario) {
                                latestTime = washingMachineTime;
                             }
                        }

        
                        document.getElementById('current-time').textContent = `Current Time: ${latestTime}`;
        
                        if (coffeeIndex < coffeeData.length) coffeeIndex++;
                        if (washingMachineIndex < washingMachineData.length) washingMachineIndex++;
        
                        setTimeout(updateValues, 200);
                    } else {
                        console.log("Tutti i dati sono stati mostrati");
                    }
                }
        
                $.getJSON('/getdata3', function(data) {
                    coffeeData = data;
                    updateValues();
                }).fail(function(jqXHR, textStatus, errorThrown) {
                    console.error("Errore nella richiesta coffee:", textStatus, errorThrown);
                    document.getElementById('coffee-consumption').textContent = 'Error fetching data';
                    document.getElementById('coffee-machine-consumption').textContent = 'Error fetching data';
                    document.getElementById('coffee-machine').classList.add('off');
                    document.getElementById('coffee-machine').classList.remove('on');
                });
        
                $.getJSON('/getdata2', function(data) {
                    washingMachineData = data;
                    updateValues();
                }).fail(function(jqXHR, textStatus, errorThrown) {
                    console.error("Errore nella richiesta washing machine:", textStatus, errorThrown);
                    document.getElementById('washing-machine-consumption').textContent = 'Error fetching data';
                    document.getElementById('washing-machine').classList.add('off');
                    document.getElementById('washing-machine').classList.remove('on');
                });
            }
        
            document.addEventListener('DOMContentLoaded', fetchData);
        </script>
